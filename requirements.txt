<script>
    // ðŸ”´ CHANGE THIS TO YOUR RENDER BACKEND URL
    const API_URL = "https://YOUR-BACKEND.onrender.com/fatigue";

    function predict() {
        const data = {
            alpha_rms: parseFloat(document.getElementById("alpha").value),
            beta_rms: parseFloat(document.getElementById("beta").value),
            gamma_rms: parseFloat(document.getElementById("gamma").value),
            delta_rms: parseFloat(document.getElementById("delta").value)
        };

        fetch(API_URL, {
            method: "POST",
            headers: {
                "Content-Type": "application/json"
            },
            body: JSON.stringify(data)
        })
        .then(response => response.json())
        .then(result => {
            const labels = [
                "ACTIVE",
                "INITIAL DROWSINESS",
                "HEAVY DROWSINESS",
                "UNCONSCIOUS"
            ];
            document.getElementById("result").innerText =
                "Predicted Fatigue Level: " + labels[result.prediction];
        })
        .catch(error => {
            document.getElementById("result").innerText =
                "Error: Unable to connect to ML backend";
        });
    }
</script>
